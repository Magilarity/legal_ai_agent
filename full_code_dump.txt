def generate_docx_from_template(template_path, context, output_path):

        doc = Document(template_path)
        for paragraph in doc.paragraphs:

        doc.save(output_path)

import os
import logging
import fitz  # PyMuPDF
from docx import Document as DocxDocument
from striprtf.striprtf import rtf_to_text
from PIL import Image
import pytesseract

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

def extract_text_from_file(path: str) -> str:
    """
    Р’РёС‚СЏРіСѓС” С‚РµРєСЃС‚ Р· С„Р°Р№Р»Сѓ Р·Р° Р№РѕРіРѕ С€Р»СЏС…РѕРј.
    РџС–РґС‚СЂРёРјСѓС” PDF, DOCX, RTF, TXT С– Р·РѕР±СЂР°Р¶РµРЅРЅСЏ (OCR).
    """
    ext = os.path.splitext(path)[1].lower()

    try:
        if ext == ".pdf":
            doc = fitz.open(path)
            pages = [page.get_text() for page in doc]
            return "\n".join(pages)

        elif ext == ".docx":
            doc = DocxDocument(path)
            return "\n".join(p.text for p in doc.paragraphs)

        elif ext == ".rtf":
            with open(path, "r", encoding="utf-8", errors="ignore") as f:
                rtf = f.read()
            return rtf_to_text(rtf)

        elif ext == ".txt":
            with open(path, "r", encoding="utf-8", errors="ignore") as f:
                return f.read()

        elif ext in [".png", ".jpg", ".jpeg", ".tiff", ".bmp", ".gif"]:
            img = Image.open(path)
            # РІРєР°Р·СѓС”РјРѕ СѓРєСЂР°С—РЅСЃСЊРєСѓ С‚Р° Р°РЅРіР»С–Р№СЃСЊРєСѓ РґР»СЏ РєСЂР°С‰РѕРіРѕ СЂРѕР·РїС–Р·РЅР°РІР°РЅРЅСЏ
            return pytesseract.image_to_string(img, lang="ukr+eng")

        else:
            logger.warning(f"РќРµРїС–РґС‚СЂРёРјСѓРІР°РЅРёР№ С„РѕСЂРјР°С‚ С„Р°Р№Р»Сѓ: {path}")
            return ""

    except Exception as e:
        logger.error(f"РџРѕРјРёР»РєР° С‡РёС‚Р°РЅРЅСЏ {ext.upper()} {path}: {e}")
        return ""

def analyze_tender_by_id(tender_id: str) -> str:
    temp_dir = "documents"
    prozorro_loader.download_documents(tender_id, temp_dir)

    result_texts = []
    signatures = []

    for filename in os.listdir(temp_dir):
        full_path = os.path.join(temp_dir, filename)

        if filename.lower().endswith((".pdf", ".docx")):
            text = document_loader.extract_text_from_file(full_path)
            if not text or not text.strip():
                result_texts.append(f"рџ“„ {filename}\n[WARNING] РџРѕСЂРѕР¶РЅС–Р№ РґРѕРєСѓРјРµРЅС‚.")
                continue

            chunks = text_splitter.split_text(text)

                result_texts.append(f"рџ“„ {filename}\n[WARNING] РќРµ РІРґР°Р»РѕСЃСЏ СЂРѕР·Р±РёС‚Рё С‚РµРєСЃС‚.")
                continue

            embeddings = embedder.embed_chunks(chunks)
            if not embeddings or not embeddings[0]:
                result_texts.append(f"рџ“„ {filename}\n[WARNING] РџРѕРјРёР»РєР° РІРµРєС‚РѕСЂРёР·Р°С†С–С—.")
                continue

            index = vector_store.create_index(embeddings)
            top_indices = vector_store.search_index(index, embeddings[0])
            # Р¤С–Р»СЊС‚СЂСѓС”РјРѕ РЅРµРІС–СЂРЅС– С–РЅРґРµРєСЃРё
            valid = [chunks[i] for i in top_indices if 0 <= i < len(chunks)]
            if not valid:
                result_texts.append(f"рџ“„ {filename}\n[WARNING] РќРµРјР° СЂРµР»РµРІР°РЅС‚РЅРёС… С„СЂР°РіРјРµРЅС‚С–РІ.")
                continue

            context = "\n---\n".join(valid)
            answer = llm_agent.generate_answer(context, f"РџСЂР°РІРѕРІРёР№ Р·РјС–СЃС‚ РґРѕРєСѓРјРµРЅС‚Р° {filename}:")

            if answer.startswith("[ERROR]"):
                result_texts.append(f"рџ“„ {filename}\n{answer}")

                result_texts.append(f"рџ“„ {filename}\n{answer}")

        elif filename.lower().endswith(".p7s"):
            info = sign_extractor.extract_signature_info(full_path)
            for signer in info:
                parts = [f"{k}: {v}" for k, v in signer.items()]

    export_utils.save_analysis_to_docx(tender_id, final, signatures)
    export_utils.save_analysis_to_pdf(tender_id, final, signatures)

    return final

def generate_answer(
    context: str,
    question: str,
    max_retries: int = 3,
    backoff_factor: float = 2.0
) -> str:
    """
    Р“РµРЅРµСЂСѓС” РІС–РґРїРѕРІС–РґСЊ С‡РµСЂРµР· OpenAI Chat API (v1.0+)
    С–Р· Р°РІС‚РѕРјР°С‚РёС‡РЅРёРј back-off Сѓ СЂР°Р·С– РїРѕРјРёР»РѕРє.
    """
    prompt = (
        f"РљРѕРЅС‚РµРєСЃС‚:\n{context}\n\n"
        f"РџРёС‚Р°РЅРЅСЏ:\n{question}\n\n"
        "Р’С–РґРїРѕРІС–РґСЊ:"
    )

    for attempt in range(1, max_retries + 1):
        try:
            response = openai.chat.completions.create(
                model=MODEL,
                messages=[{"role": "user", "content": prompt}],
                temperature=0.3
            )
            return response.choices[0].message.content.strip()

        except Exception as e:
            wait = backoff_factor ** (attempt - 1)
            logging.warning(
                f"[Attempt {attempt}/{max_retries}] РџРѕРјРёР»РєР° РїСЂРё РІРёРєР»РёРєСѓ OpenAI: {e}. "
                f"Р§РµРєР°СЋ {wait} СЃРµРє..."
            )
            time.sleep(wait)

    raise RuntimeError("РќРµ РІРґР°Р»РѕСЃСЏ РѕС‚СЂРёРјР°С‚Рё РІС–РґРїРѕРІС–РґСЊ РІС–Рґ OpenAI РїС–СЃР»СЏ РєС–Р»СЊРєРѕС… СЃРїСЂРѕР±.")

import asn1crypto.cms
import asn1crypto.pem
import logging

logging.basicConfig(level=logging.INFO, format='%(levelname)s: %(message)s')

def extract_signature_info(file_path: str) -> list:
    """
    РџРѕРІРµСЂС‚Р°С” СЃРїРёСЃРѕРє СЃР»РѕРІРЅРёРєС–РІ С–Р· РїРѕР»СЏРјРё issuer/serial_number.
    РЇРєС‰Рѕ С„Р°Р№Р» РЅРµ РІР°Р»С–РґРЅРёР№ PKCS#7, РїРѕРІРµСЂС‚Р°С” РїСѓСЃС‚РёР№ СЃРїРёСЃРѕРє.
    """
    try:
        with open(file_path, 'rb') as f:
            raw_data = f.read()

        # РЇРєС‰Рѕ PEM-РѕР±РіРѕСЂС‚РєР°, Р·РЅС–РјР°С”РјРѕ С—С—
        if asn1crypto.pem.detect(raw_data):
            _, _, raw_data = asn1crypto.pem.unarmor(raw_data)

        content_info = asn1crypto.cms.ContentInfo.load(raw_data)
        signer_infos = content_info['content']['signer_infos']
        certs = content_info['content'].get('certificates', [])

        result = []
        for signer in signer_infos:
            sid = signer['sid']
            serial = sid.chosen['serial_number'].native
            # Р—РЅР°С…РѕРґРёРјРѕ СЃРµСЂС‚РёС„С–РєР°С‚ Р·Р° СЃРµСЂС–Р№РЅРёРєРѕРј
            signer_cert = None
            for cert in certs:
                cert_obj = cert if isinstance(cert, asn1crypto.x509.Certificate) else asn1crypto.x509.Certificate.load(cert.dump())
                if cert_obj.serial_number == serial:
                    signer_cert = cert_obj
                    break
            issuer = signer_cert.subject.human_friendly if signer_cert else "Рќ/Р”"
            result.append({
                'issuer': issuer,
                'serial_number': serial
            })

        return result

    except (ValueError, TypeError) as e:
        # РќРµ РІР°Р»С–РґРЅРёР№ PKCS#7 Р°Р±Рѕ РЅРµ РІРґР°Р»РѕСЃСЏ СЂРѕР·РїС–Р·РЅР°С‚Рё вЂ” РїСЂРѕСЃС‚Рѕ РїРѕРІРµСЂС‚Р°С”РјРѕ РїСѓСЃС‚РёР№ СЃРїРёСЃРѕРє
        logging.warning(f"РќРµ РІР°Р»С–РґРЅРёР№ .p7s Р°Р±Рѕ РїРѕРјРёР»РєР° РїР°СЂСЃРёРЅРіСѓ: {e}")
        return []
    except Exception as e:
        logging.error(f"РџРѕРјРёР»РєР° РїСЂРё РІРёС‚СЏРіСѓРІР°РЅРЅС– РїС–РґРїРёСЃСѓ: {e}")

import numpy as np

def create_index(embeddings):
    dim = len(embeddings[0])
    index = faiss.IndexFlatL2(dim)
    index.add(np.array(embeddings).astype('float32'))
    return index

def search_index(index, query_vector, top_k=3):
    D, I = index.search(np.array([query_vector]).astype('float32'), top_k)

import os
from pathlib import Path

from sqlalchemy import text
from db.schema import Session, Document, ENGINE

import fitz       # PyMuPDF
import docx       # python-docx

def extract_text(path: str) -> str:
    ext = Path(path).suffix.lower()
    try:
        if ext == ".pdf":
            text_chunks = []
            with fitz.open(path) as pdf:
                for page in pdf:
                    text_chunks.append(page.get_text())
            return "\n".join(text_chunks)
        elif ext in (".docx",):
            doc = docx.Document(path)
            return "\n".join(p.text for p in doc.paragraphs)
        else:
            # РґР»СЏ С–РЅС€РёС… С„РѕСЂРјР°С‚С–РІ РјРѕР¶РЅР° РґРѕРґР°С‚Рё Р»РѕРіС–РєРё, РїРѕРєРё вЂ“ РїСѓСЃС‚РёР№ СЂСЏРґРѕРє
            return ""
    except Exception as e:
        print(f"[warn] РїРѕРјРёР»РєР° РІРёС‚СЏРіСѓ С‚РµРєСЃС‚Сѓ Р· {path}: {e}")
        return ""

def ingest_all(downloads_folder="downloads"):
    session = Session()
    for tender_dir in Path(downloads_folder).iterdir():
        if not tender_dir.is_dir():
            continue
        tender_id = tender_dir.name
        for file in tender_dir.rglob("*"):
            if not file.is_file():
                continue
            content = extract_text(str(file))
            if content.strip():
                doc = Document(
                    tender=tender_id,
                    title=file.name,
                    content=content
                )
                session.add(doc)
    session.commit()

    # rebuild FTS index
    with ENGINE.connect() as conn:
        conn.execute(text("INSERT INTO documents_fts(documents_fts) VALUES('rebuild')"))

    print("вњ… Ingestion complete.")

if __name__ == "__main__":
    ingest_all()

def fetch_with_retry(url: str) -> requests.Response:
    """GET Р· РµРєСЃРїРѕРЅРµРЅС†С–Р°Р»СЊРЅРёРј Р±РµРє-РѕС„РѕРј Сѓ СЂР°Р·С– 429."""
    backoffs = [1, 2, 4]
    for b in backoffs:
        r = session.get(url, timeout=30)
        if r.status_code == 429:
            time.sleep(b)
            continue
        r.raise_for_status()
        return r
    # С„С–РЅР°Р»СЊРЅР° СЃРїСЂРѕР±Р°
    r = session.get(url, timeout=30)
    r.raise_for_status()
    return r


def save_doc(doc: dict, folder: str, logger: Callable[[str], None] = print) -> int:
    """Р—Р±РµСЂС–РіР°С” РѕРґРёРЅ РґРѕРєСѓРјРµРЅС‚, РїРѕРІРµСЂС‚Р°С” 1 СЏРєС‰Рѕ СѓСЃРїС–С€РЅРѕ, С–РЅР°РєС€Рµ 0."""
    title = doc.get("title") or f"doc_{int(time.time())}"
    filename = sanitize_filename(title)
    url = doc.get("url")
    if not url:
        logger(f"[warn] РџСЂРѕРїСѓС‰РµРЅРѕ вЂ“ РЅРµРјР°С” URL: {filename}")
        return 0

    os.makedirs(folder, exist_ok=True)
    path = os.path.join(folder, filename)
    if os.path.exists(path):
        logger(f"[skip] РІР¶Рµ С–СЃРЅСѓС”: {filename}")
        return 0

    try:
        r = fetch_with_retry(url)
        with open(path, "wb") as f:
            f.write(r.content)
        logger(f"[v] Р·Р°РІР°РЅС‚Р°Р¶РµРЅРѕ: {filename}")
        return 1
    except Exception as e:
        logger(f"[err] РЅРµ РІРґР°Р»РѕСЃСЏ Р·Р°РІР°РЅС‚Р°Р¶РёС‚Рё {filename}: {e}")
        return 0

    """Р РµРєСѓСЂСЃРёРІРЅРѕ Р·РЅР°С…РѕРґРёС‚СЊ СѓСЃС– dict С–Р· 'url' С– 'title'."""
    if isinstance(item, dict):
        if "url" in item and "title" in item:
            yield item
        for v in item.values():
            yield from extract_all_docs(v)
    elif isinstance(item, list):
        for elem in item:
            yield from extract_all_docs(elem)

import sys
from sqlalchemy import text
from db.schema import Session

def ft_search(section_name: str, fts_table: str, data_table: str, title_col: str, query: str, limit: int):
    print(f"\n=== {section_name} ===")
    session = Session()
    sql = text(f"""
      SELECT d.id AS id,
             d.{title_col} AS title,
             snippet({fts_table}, -1, '<b>','</b>','...',10) AS snippet
      FROM {fts_table}
      JOIN {data_table} AS d ON {fts_table}.rowid = d.id
      WHERE {fts_table} MATCH :q
      LIMIT :lim
    """)
    rows = session.execute(sql, {"q": query, "lim": limit})
    found = False
    for r in rows:
        found = True
        print(f"{r.id}\t{r.title}\n  вЂ¦{r.snippet}\n")
    if not found:
        print("РќС–С‡РѕРіРѕ РЅРµ Р·РЅР°Р№РґРµРЅРѕ.")
